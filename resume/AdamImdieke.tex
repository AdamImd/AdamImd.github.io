\section{Projects}
\runsubsection{Tactile Skin for Spot}
\descript{| Research Project}
\location{\faIcon{link} \href{https://adamimd.github.io/projects/Tactile_Skin/}{Project Page} | May 2025–Present}
\begin{tightemize}
\item A Novel, Low-cost 3D printed tactile skin for robotic arms to enhance whole-body environmental contact sensing.
\item Leverages high Degree of Freedom robots to condition Inverse Kinematics null spaces to satisfy contact constraints.
\item Hardware development for real-time Contact aware Inverse Kinematics (ContactIK) enables both contact-avoidance and contact-embracing behaviors in manipulation.
\end{tightemize}
\sectionsep
